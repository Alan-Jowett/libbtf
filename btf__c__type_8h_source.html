42 typedef struct _btf_type {
43  uint32_t name_off;
44  /* "info" bits arrangement
45  * bits 0-15: vlen (e.g. # of struct's members)
46  * bits 16-23: unused
47  * bits 24-28: kind (e.g. int, ptr, array...etc)
48  * bits 29-30: unused
49  * bit 31: kind_flag, currently used by
50  * struct, union and fwd
51  */
52  uint32_t info;
53  /* "size" is used by INT, ENUM, STRUCT and UNION.
54  * "size" tells the size of the type it is describing.
55  *
56  * "type" is used by PTR, TYPEDEF, VOLATILE, CONST, RESTRICT,
57  * FUNC, FUNC_PROTO, DECL_TAG and TYPE_TAG.
58  * "type" is a type_id referring to another type.
59  */
60  union {
61  uint32_t size;
62  uint32_t type;
63  };
64 } btf_type_t;
